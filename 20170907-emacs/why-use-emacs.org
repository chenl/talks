** Hard Customization

#+BEGIN_SRC elisp
(add-hook 'org-mode-hook
   #'(lambda ()
      ;; supporting modes
      (require 'org-bullets)
      (org-bullets-mode 1)
      ;; key bindings: "C-c !" is a leader for fly-check
      (local-set-key (kbd "C-c 1")
         #'org-time-stamp-inactive)
      ;; org-reveal
      (require 'ox-reveal)
      (setq org-reveal-root (concat
         "file://" (getenv "HOME") "/opt/reveal.js"))))
#+END_SRC
